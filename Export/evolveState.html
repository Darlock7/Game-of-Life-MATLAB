function out = evolveState(R,C,m,n)
 
x = 1; %preallocation
y = 1;
surviveCell = {['Na']}; %using Na as preallocation to avoid 'unrecognized variable' - needs to be a non-number to prevent filling in false cells
makeDeadCell = {['Na']}; %must be two letters so that when compared to real [R,C] arrays have compatible sizes for comparison
 
for i = 1:length(R)
if survive(R,C, R(i), C(i))
surviveCell{x} = [R(i), C(i)]; %storing all surviving cells in an array to use later
x = x + 1;
else
makeDeadCell{y} = [R(i), C(i)]; %storing all dying cells in an array to use later
y = y + 1;
end
end
 
z = 1; %preallocation
awakenCell = {['Na']};
for j = 0:(m-1)
for k = 0:(n-1)
if awakens(R,C,j, k)
awakenCell{z} = [j, k]; %storing awakening cells in an array to use later
z = z + 1;
else
continue;
end
end
end
 
%killing cells using stored dying cells
for i = 1:length(makeDeadCell)
if makeDeadCell{1} == 'Na'
continue;
else
rectangle('Position',[makeDeadCell{i}(2), makeDeadCell{i}(1),1,1],'FaceColor','w','EdgeColor', [0.5, 0.5, 0.5]);
end
end
 
%filling cells using stored awakening cells
for j = 1:length(awakenCell)
if awakenCell{1} == 'Na'
continue;
else
rectangle('Position',[awakenCell{j}(2), awakenCell{j}(1),1,1],'FaceColor','k','EdgeColor', [0.5, 0.5, 0.5]);
end
end
 
 
newLiveCells = [surviveCell, awakenCell]; %creating a cell array containing the new live cells
for i = 1:length(newLiveCells)
if newLiveCells{i} == 'Na'
newLiveCells{i} = [0 0];
end
end
Rnew = zeros(1,length(newLiveCells)); %preallocation
Cnew = zeros(1,length(newLiveCells));
for j = 1:length(newLiveCells)
Rnew(j) = newLiveCells{j}(1); %rows of new live cells
Cnew(j) = newLiveCells{j}(2); %columns of new live cells
end
out = {Rnew, Cnew};
end
