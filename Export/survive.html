function out = survive(R,C,m,n)
z = [m n];
 
count = 0; %initializing count
RCheck = [1 0 -1]; %pattern for checking surrounding cells
CCheck = [-1 0 1];
 
%checking surrounding cells and increasing count if liveCell is around
%cell
for i = 1:3
for j = 1:3
if [(m + RCheck(i)), (n + CCheck(j))] == z %avoids counting cell we're looking at
continue;
elseif liveCell(R,C, (m + RCheck(i)), (n + CCheck(j)))
count = count + 1;
end
end
end
 
%if count equals exactly 2 or 3 out returns true (cell survives) other
%false (cell dies)
if (count == 2) | (count ==3)
out = true;
else
out = false;
end
 
end
