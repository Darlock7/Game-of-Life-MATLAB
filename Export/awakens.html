function out = awakens(R,C,m,n)
a = [m n];
count = 0;
RCheck = [1 0 -1];
CCheck = [-1 0 1];
 
if liveCell(R,C,m,n) %if cell we're looking at is a live cell move to next cell
;
else %checking each surrounding cell and increasing count if live cell is near
for i = 1:3
for j = 1:3
if [(m + RCheck(i)), (n + CCheck(j))] == a %avoids counting cell we're looking at
continue;
elseif liveCell(R,C,(m + RCheck(i)), (n + CCheck(j)))
count = count + 1;
end
end
end
%if three live cells surround this cell evaluate to true(awaken)
% else evaluate to false (do nothing)
end
if count == 3
out = true;
else
out = false;
end
end
